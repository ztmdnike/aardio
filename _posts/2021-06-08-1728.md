```js
import console;
var s = "a中";
ws=loadcodex(string.format("return '%s'u;", s))//执行代码
ws=eval(string.format("'%s'u;", s))//返回一个代码结果
ws=eval("'"++s++"'u;")
ws='a中'u
console.log(ws)
console.log(#ws/2)
for(i=1;#ws/2 ){
    console.log("Unicode字节码", ws[ i ], "Unicode字符",ws[[ i ]] )
}
console.pause(true);
```
